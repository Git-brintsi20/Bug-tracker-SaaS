```javascript
const getCachedBugs = async (orgId, filters) => {
  const cacheKey = `bugs:${orgId}:${JSON.stringify(filters)}`;
  
  // Try cache first
  const cached = await redis.get(cacheKey);
  if (cached) return JSON.parse(cached);
  
  // Fetch from DB
  const bugs = await prisma.bug.findMany({
    where: { organizationId: orgId, ...filters }
  });
  
  // Cache for 5 minutes
  await redis.setex(cacheKey, 300, JSON.stringify(bugs));
  
  return bugs;
};

// Invalidate on update
```

```typescript
'use client';

import { useEffect, useState } from 'react';
import { socket } from '@/lib/socket';
import { Bug } from '@/types';

export function BugList({ organizationId }: { organizationId: string }) {
  const [bugs, setBugs] = useState<Bug[]>([]);

  useEffect(() => {
    // Join organization room
    socket.emit('join-organization', { organizationId });

    // Listen for bug updates
    socket.on('bug-created', (bug: Bug) => {
      setBugs(prev => [bug, ...prev]);
    });

    socket.on('bug-updated', ({ bugId, changes }) => {
      setBugs(prev => prev.map(bug => 
        bug.id === bugId ? { ...bug, ...changes } : bug
      ));
    });

    socket.on('bug-deleted', ({ bugId }) => {
      setBugs(prev => prev.filter(bug => bug.id !== bugId));
    });

    return () => {
      socket.emit('leave-organization', { organizationId });
      socket.off('bug-created');
      socket.off('bug-updated');
      socket.off('bug-deleted');
    };
  }, [organizationId]);

  return (
    <div>
      {bugs.map(bug => (
        <BugCard key={bug.id} bug={bug} />
      ))}
    </div>
  );
}
```

```javascript
// services/bug-service/src/utils/cache.js
const redis = require('./redis');
const crypto = require('crypto');

// Generate cache key with filters
const generateCacheKey = (orgId, filters) => {
  const filterHash = crypto
    .createHash('md5')
    .update(JSON.stringify(filters))
    .digest('hex');
  
  return `bugs:${orgId}:list:${filterHash}`;
};

// Get bugs with caching
const getBugsWithCache = async (orgId, filters) => {
  const cacheKey = generateCacheKey(orgId, filters);
  
  // Try cache
  const cached = await redis.get(cacheKey);
  if (cached) {
    console.log('Cache hit:', cacheKey);
    return JSON.parse(cached);
  }
  
  // Fetch from DB
  const bugs = await prisma.bug.findMany({
    where: {
      organizationId: orgId,
      ...filters
    },
    include: {
      creator: true,
      assignee: true
    }
  });
  
  // Cache for 5 minutes
  await redis.setex(cacheKey, 300, JSON.stringify(bugs));
  console.log('Cache miss, stored:', cacheKey);
  
  return bugs;
};

// Invalidate cache on update
const invalidateBugCache = async (orgId) => {
  const pattern = `bugs:${orgId}:list:*`;
  const keys = await redis.keys(pattern);
  
  if (keys.length > 0) {
    await redis.del(...keys);
    console.log('Invalidated cache:', keys.length, 'keys');
  }
};

module.exports = {
  getBugsWithCache,
  invalidateBugCache
};
```

### ğŸ’» Middleware Example
```javascript
// services/auth-service/src/middleware/auth.js
const jwt = require('jsonwebtoken');

const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'Access token required' });
  }

  jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
    if (err) {
      return res.status(403).json({ error: 'Invalid or expired token' });
    }
    
    req.user = user;
    next();
  });
};

// RBAC middleware
const requireRole = (roles) => {
  return async (req, res, next) => {
    const { organizationId } = req.params;
    
    const member = await prisma.organizationMember.findUnique({
      where: {
        userId_organizationId: {
          userId: req.user.id,
          organizationId
        }
      }
    });

    if (!member || !roles.includes(member.role)) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }

    req.userRole = member.role;
    next();
  };
};

module.exports = { authenticateToken, requireRole };
```
